信息数据包

玩家加入服务

{
    "type":"pack",
    "cause": "join",
    "params": {
        "sender": "ExamplePlayer",
        "xuid": "",
        "ip": "target's ip address"
    }
}

玩家退出服务器

{
    "type":"pack",
    "cause": "left",
    "params": {
        "sender": "ExamplePlayer",
        "xuid": ""
    }
}

玩家使用命令

{
    "type":"pack",
    "cause": "cmd",
    "params": {
        "sender": "ExamplePlayer",
        "cmd": "/kill @s"
    }
}

    "type":"pack",
